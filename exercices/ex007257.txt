\exercice{7257, mourougane, 2021/08/10}
\enonce
\begin{enumerate}

\item Résoudre dans $\Rr$ l'équation $\sqrt{3} \cos(x) + \sin(x) = \sqrt{2}$.

\item \`A l'aide une méthode similaire, résoudre l'équation $\cos(x) + \sin(x)=1$.

\end{enumerate}
\finenonce
\noindication
\nocorrection
\finexercice
