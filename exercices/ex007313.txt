\exercice{7313, mourougane, 2021/08/10}
\enonce
\begin{enumerate}
 \item Dans l'anneau $\Z /3 \Z$, calculer $(a+b)^3$ où $a$ et $b$ sont
deux éléments.

 \item Dans l'anneau $\Z /6 \Z$, calculer $(a+b)^6$ où $a$ et $b$ sont
deux éléments.

 \item Dans l'anneau $\Z /7 \Z$, calculer $(a+b)^7$ où $a$ et $b$ sont
deux éléments.
\end{enumerate}
\finenonce
\noindication
\nocorrection
\finexercice
