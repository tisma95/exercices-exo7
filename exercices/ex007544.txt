\exercice{7544, mourougane, 2021/08/10}
\enonce[Calcul de rayon de convergence]
\begin{enumerate}
\item Déterminer le rayon de convergence de la série $\sum 2^{2n}z^{2n}$.
\item Déterminer la limite de $\frac{2^{2(n+1)}}{2^{2n}}$.
\end{enumerate}
\finenonce
\noindication
\nocorrection
\finexercice
