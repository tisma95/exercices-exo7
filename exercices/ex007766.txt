\exercice{7766, mourougane, 2021/08/11}
\enonce
\begin{enumerate}
\item Si $c$ est le cycle $(1,2,3,4,5,6)$, $c^2$ est-il un cycle ?
\item
Si $c$ est un cycle de $\mathcal{S}_n$ d'ordre $l$ et $k$ un entier naturel,
calculer l'ordre de $c^k$.
\end{enumerate}
\finenonce
\noindication
\nocorrection
\finexercice
