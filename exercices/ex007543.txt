\exercice{7543, mourougane, 2021/08/10}
\enonce[Un exemple]
\begin{enumerate}
\item Montrer que la série $\sum_{n\geq 1}\frac{(-1)^n}{z+n}$ converge de façon compacte sur $\Cc-\{-1,-2,-3\cdots\}$.
\item La convergence est-elle normale ?
\end{enumerate}
\finenonce
\noindication
\nocorrection
\finexercice
